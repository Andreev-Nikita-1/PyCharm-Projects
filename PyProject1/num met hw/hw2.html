def add(d1, d2, d3):
    return (d1 - d2)**2 / (d1 + d3 - 2 * d2)


def eitken(sqs):
    newsqs = list(sqs[:2])
    for i in range(2, len(sqs)):
        a = add(sqs[i], sqs[i - 1], sqs[i - 2])
        newsqs.append(sqs[i] - a)
    return newsqs
