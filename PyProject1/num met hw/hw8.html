def euler(f, f0, N, interval=(0, 10)):
    a, b = interval
    h = (b - a) / N
    xs = np.arange(a, b + h / 2, h)
    ys = [f0]
    for x in xs[:-1]:
        y_next = ys[-1] + f(x, ys[-1]) * h
        ys.append(y_next)
    return xs, ys

def runge_kutta(f, f0, N, interval=(0, 10), beta=1 / 2):
    a, b = interval
    h = (b - a) / N
    xs = np.arange(a, b + h / 2, h)
    ys = [f0]
    for x in xs[:-1]:
        y = ys[-1]
        fxy = f(x, y)
        y_next = y + h * ((1 - beta) * fxy + beta * f(x + h /
                                                      (2 * beta), y + h /
                                                      (2 * beta) * fxy))
        ys.append(y_next)
    return xs, ys
