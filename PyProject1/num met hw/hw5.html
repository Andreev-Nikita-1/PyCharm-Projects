$w(x) = \sum\limits_3^\infty\frac{1}{k^2-k-x}$

$I = \int\limits_0^3 w(x) d x$

$\int\frac{1}{k^2-k-x} = \log \left(\frac{k^2 - k}{k^2-k-3}\right) = \log \left(1 +\frac{3}{k^2-k-3}\right) = \frac{3}{k^2-k-3} +O(\frac{1}{k^4}) \approx \frac{3}{k^2}$

$\log \left(1 +\frac{3}{k^2-k-3}\right) - \frac{3}{k^2} - \frac{3}{k^3}= O(\frac{1}{k^4})$

$\sum\limits_{N+1}^\infty \log \left(1 +\frac{3}{k^2-k-3}\right) - \frac{3}{k^2} - \frac{3}{k^3}= O(\frac{1}{N^3})$

def w(x, N=int(1e+4)):
    # N = 1e+4, error ~= 1e-16
    def term(k):
        return (k * (1 + 2 * x) + (1 + x) * x) / (k**4 * (k**2 - k - x))

    const = np.pi**2 / 6 + scipy.special.zeta(
        3) - 2 - 1 / 4 - 1 / 8 + (1 + x) * (scipy.special.zeta(4) - 1 - 1 / 16)
    return const + np.sum([term(k) for k in range(3, N + 1)][::-1])

def simpson(f, N):
    h = 3 / N

    def coeff(k):
        if k == 0 or k == N:
            return 1
        elif k == int(k):
            return 2
        else:
            return 4

    return h / 6 * np.sum(
        [coeff(k) * f(k * h) for k in np.arange(0, N + 1 / 2, 1 / 2)][::-1])

def newton(deg, a, b, eps=1e-15):
    f = scipy.special.legendre(deg)
    #     df = f.deriv(1)
    #     почему то начинает считать неправильно при степенях порядка 37
    df = lambda z: deg / (1 - z**2) * (scipy.special.legendre(deg - 1)
                                       (z) - z * f(z))
    x = (a + b) / 2
    while True:
        fx = f(x)
        x1 = x - fx / df(x)
        if x1 < a or x1 > b:
            fa, fb = f(a), f(b)
            if fx * fa < 0:
                x1 = (a + x) / 2
            elif fx * fb < 0:
                x1 = (b + x) / 2
        if np.abs(x - x1) < eps:
            break
        x = x1
    return x

def find_roots(deg, method=newton):
    roots_legendre = {0: np.array([])}
    for d in range(1, deg + 1):
        roots_bounds = np.hstack([[-1], roots_legendre[d - 1], [1]])
        roots = []
        for a, b in zip(roots_bounds, roots_bounds[1:]):
            roots.append(method(d, a, b))
        roots_legendre[d] = np.array(roots)
    return roots_legendre

def weights(deg, xs):
    f = scipy.special.legendre(deg)
    # если считать производную численно или через f.deriv(1), 
    # то при достижении точности  порядка 1e-10 что-то ломается и результат не улучшается
    df = lambda z: deg / (1 - z**2) * (scipy.special.legendre(deg - 1)
                                       (z) - z * f(z))
    ans = [2 / (1 - x**2) / df(x)**2 for x in xs]
    return ans

def gauss(f, zs, ws):
